// core/orchestrator.js
const { validateIntent } = require('./adapters/validators');
const dropframeAdapter = require('./adapters/dropframe-adapter');
const grindlineAdapter = require('./adapters/grindline-adapter');

class RadiantOrchestrator {
  constructor(config) {
    this.config = config;
    this.activeRequests = new Map();
  }

  async handleIntent(message) {
    const validation = validateIntent(message);
    if (!validation.valid) {
      return this.createErrorResponse(message.id, validation.errors);
    }

    const { mode, description, inputs, constraints } = message.payload;

    // Route based on intent mode
    switch (mode) {
      case 'story':
        return await this.handleStoryIntent(message);
      case 'content':
        return await this.handleContentIntent(message);
      case 'analysis':
        return await this.handleAnalysisIntent(message);

        return this.createErrorResponse(message.id, `Unknown mode: ${mode}`);
    }
  }

    // Delegate to DropFrame for story operations
    try {
      const response = await dropframeAdapter.transformStory(message);
      return response;
    } catch (error) {
      return this.createErrorResponse(message.id, error.message);

async function transformStory(message) {
  const endpoint = process.env.DROPFRAME_ENDPOINT || 'http://localhost:3001';

  try {
    const response = await fetch(`${endpoint}/transform`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(message),
      signal: AbortSignal.timeout(30000), // 30s timeout
    });

    if (!response.ok) {
      throw new Error(`DropFrame error: ${response.statusText}`);
    }

    return await response.json();
  } catch (error) {
    if (error.name === 'AbortError') {
      throw new Error('DropFrame request timeout');
    }
    throw error;
  }
}

**6. Implement API Routes**

Create Next.js API routes for protocol execution:
```typescript
// apps/web/src/app/api/radiant/intent/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { MessageEnvelopeSchema } from '@radiant/types/validators';
import RadiantOrchestrator from '@/../core/orchestrator';

const orchestrator = new RadiantOrchestrator(/* config */);

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();

    // Validate message envelope
    const validation = MessageEnvelopeSchema.safeParse(body);
    if (!validation.success) {
      return NextResponse.json(
        { error: 'Invalid message format', details: validation.error },
        { status: 400 }
      );
    }

    // Process intent
    const response = await orchestrator.handleIntent(validation.data);
    return NextResponse.json(response);

  } catch (error) {
    return NextResponse.json(
      { error: 'Internal server error', message: error.message },
      { status: 500 }
    );
  }
}
```
